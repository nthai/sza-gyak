    \begin{tikzpicture}
        [
            vertex/.style={draw,circle,inner sep = 0em, minimum size = 0.5em, fill=white},

        \node[vertex, label = {[nodelabel]right : t}] (t) at (3, 0) {};

        \draw[-{>[length=5, width=5]}] (s) to node[edgelabel,sloped] {$1(3)$} (b);
